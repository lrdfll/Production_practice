Запрос, вычисляющий полную стоимость материалов на весь объем заказа согласно спецификации.

WITH MaterialCosts AS (
    SELECT 
        s."id_Продукции",
        SUM(ss."Количество" * c."Цена") as CostOfMaterialsPerUnit
    FROM "Спецификация" s
    JOIN "СоставСпецификации" ss ON s."id_Спецификации" = ss."id_Спецификации"
    JOIN "Цены" c ON ss."id_Номенклатуры" = c."id_Номенклатуры"
    JOIN "ТипЦены" tc ON c."id_ТипаЦены" = tc."id_ТипаЦены"
    WHERE tc."НаименованиеТипаЦены" ILIKE '%закупочная%'
    GROUP BY s."id_Продукции"
)
SELECT 
    z."id_Заказа",
    z."ДатаЗаказа",
    k."Наименование" as "Покупатель",
    n."НаименованиеТовара" as "Продукция",
    sz."Количество" as "Кол_во_в_заказе",
    sz."Цена" as "Цена_продажи",
    (sz."Количество" * sz."Цена") as "Сумма_заказа_общая",
    ROUND(sz."Количество" * COALESCE(mc.CostOfMaterialsPerUnit, 0), 2) as "Полная_стоимость_материалов"
FROM "ЗаказыПокупателей" z
JOIN "Контрагенты" k ON z."id_Контрагента" = k."id_Контрагента"
JOIN "СоставЗаказа" sz ON z."id_Заказа" = sz."id_Заказа"
JOIN "Номенклатура" n ON sz."id_Номенклатуры" = n."id_Номенклатуры"
LEFT JOIN MaterialCosts mc ON n."id_Номенклатуры" = mc."id_Продукции"
ORDER BY z."id_Заказа";
